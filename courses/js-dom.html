Pelajaran 1: Pengenalan DOM

DOM (Document Object Model) adalah representasi struktur HTML sebagai objek yang dapat dimanipulasi dengan JavaScript.

Apa itu DOM?

DOM memungkinkan JavaScript untuk:

  • Mengubah konten HTML
  • Mengubah atribut HTML
  • Mengubah style CSS
  • Menambah atau menghapus elemen
  • Merespons event HTML

Struktur DOM Tree

<!-- HTML -->
<html>
  <head>
    <title>My Page</title>
  </head>
  <body>
    <h1 id="title">Hello World</h1>
    <p class="text">This is a paragraph.</p>
  </body>
</html>

Document Object

// Mengakses document
console.log(document); // Seluruh dokumen HTML
console.log(document.title); // Judul halaman
console.log(document.URL); // URL halaman
console.log(document.body); // Element body

// Mengubah title
document.title = 'Judul Baru';

Window Object

// Window adalah objek global
console.log(window.innerWidth); // Lebar window
console.log(window.innerHeight); // Tinggi window

// Alert, confirm, prompt
window.alert('Hello!');
let konfirmasi = window.confirm('Apakah Anda yakin?');
let input = window.prompt('Masukkan nama Anda:');

Pelajaran 2: Selecting Elements

Sebelum memanipulasi elemen, kita perlu memilih elemen tersebut terlebih dahulu.

getElementById

// HTML: <h1 id="title">Hello</h1>
let title = document.getElementById('title');
console.log(title); // <h1 id="title">Hello</h1>
console.log(title.textContent); // "Hello"

getElementsByClassName

// HTML: <p class="text">Paragraph 1</p>
//       <p class="text">Paragraph 2</p>
let paragraphs = document.getElementsByClassName('text');
console.log(paragraphs); // HTMLCollection [p.text, p.text]
console.log(paragraphs[0]); // Elemen pertama
console.log(paragraphs.length); // 2

// Loop through collection
for (let p of paragraphs) {
    console.log(p.textContent);
}

getElementsByTagName

// Pilih semua elemen dengan tag tertentu
let allParagraphs = document.getElementsByTagName('p');
let allDivs = document.getElementsByTagName('div');

// Pilih semua elemen
let allElements = document.getElementsByTagName('*');

querySelector dan querySelectorAll

// querySelector - pilih elemen pertama yang cocok
let title = document.querySelector('#title'); // by ID
let firstText = document.querySelector('.text'); // by class
let firstP = document.querySelector('p'); // by tag
let specific = document.querySelector('div.container p.highlight');

// querySelectorAll - pilih semua elemen yang cocok
let allTexts = document.querySelectorAll('.text');
let allPs = document.querySelectorAll('p');

// Advanced selectors
let evenPs = document.querySelectorAll('p:nth-child(even)');
let firstChild = document.querySelector('div > p:first-child');

Pelajaran 3: Manipulasi Content

Setelah memilih elemen, kita dapat mengubah konten, atribut, dan properti elemen tersebut.

Mengubah Text Content

let title = document.getElementById('title');

// textContent - hanya teks, tanpa HTML
title.textContent = 'Judul Baru';
console.log(title.textContent); // "Judul Baru"

// innerHTML - termasuk HTML tags
title.innerHTML = '<strong>Judul Tebal</strong>';

// innerText - teks yang terlihat (respects styling)
title.innerText = 'Teks Terlihat';

Manipulasi Atribut

let image = document.querySelector('img');

// getAttribute - mendapatkan nilai atribut
let src = image.getAttribute('src');
let alt = image.getAttribute('alt');

// setAttribute - mengatur nilai atribut
image.setAttribute('src', 'new-image.jpg');
image.setAttribute('alt', 'Gambar baru');

// removeAttribute - menghapus atribut
image.removeAttribute('title');

// hasAttribute - cek apakah atribut ada
if (image.hasAttribute('data-id')) {
    console.log('Element memiliki data-id');
}

Properti Elemen

let input = document.querySelector('input');
let link = document.querySelector('a');

// Input properties
input.value = 'Nilai baru';
input.placeholder = 'Masukkan teks...';
input.disabled = true;
input.checked = true; // untuk checkbox/radio

// Link properties
link.href = 'https://example.com';
link.target = '_blank';

// Common properties
element.id = 'new-id';
element.className = 'new-class';
element.title = 'Tooltip text';

Class Manipulation

let element = document.querySelector('.my-element');

// classList methods
element.classList.add('new-class'); // tambah class
element.classList.remove('old-class'); // hapus class
element.classList.toggle('active'); // toggle class
element.classList.replace('old', 'new'); // ganti class

// Check if class exists
if (element.classList.contains('active')) {
    console.log('Element memiliki class active');
}

// Get all classes
console.log(element.classList); // DOMTokenList

Pelajaran 4: Manipulasi Style

JavaScript dapat mengubah style CSS elemen secara langsung atau melalui class.

Inline Styles

let element = document.getElementById('myElement');

// Mengubah style individual
element.style.color = 'red';
element.style.backgroundColor = 'yellow';
element.style.fontSize = '20px';
element.style.margin = '10px';
element.style.display = 'none'; // sembunyikan
element.style.display = 'block'; // tampilkan

// Camel case untuk property dengan dash
element.style.borderRadius = '5px'; // border-radius
element.style.textAlign = 'center'; // text-align

CSS Text dan cssText

let element = document.querySelector('.box');

// cssText - set multiple styles at once
element.style.cssText = 'color: blue; background: yellow; padding: 10px;';

// Atau dengan template literal
element.style.cssText = `
    color: ${textColor};
    background-color: ${bgColor};
    padding: ${padding}px;
`;

// Mendapatkan computed style
let computedStyle = window.getComputedStyle(element);
console.log(computedStyle.color); // rgb(0, 0, 255)
console.log(computedStyle.fontSize); // 16px

Show/Hide Elements

let element = document.querySelector('.toggle-element');

// Cara 1: display
function hideElement() {
    element.style.display = 'none';
}

function showElement() {
    element.style.display = 'block'; // atau 'inline', 'flex', dll
}

// Cara 2: visibility
function hideElementKeepSpace() {
    element.style.visibility = 'hidden';
}

function showElementKeepSpace() {
    element.style.visibility = 'visible';
}

// Cara 3: opacity
function fadeOut() {
    element.style.opacity = '0';
}

function fadeIn() {
    element.style.opacity = '1';
}

Dynamic Styling

// Animasi sederhana dengan JavaScript
function animateWidth(element, targetWidth) {
    let currentWidth = 0;
    let interval = setInterval(() => {
        if (currentWidth >= targetWidth) {
            clearInterval(interval);
        } else {
            currentWidth += 5;
            element.style.width = currentWidth + 'px';
        }
    }, 10);
}

// Responsive styling
function updateStyleBasedOnScreenSize() {
    let element = document.querySelector('.responsive-element');
    
    if (window.innerWidth < 768) {
        element.style.fontSize = '14px';
        element.style.padding = '5px';
    } else {
        element.style.fontSize = '16px';
        element.style.padding = '10px';
    }
}

// Jalankan saat window resize
window.addEventListener('resize', updateStyleBasedOnScreenSize);

Pelajaran 5: Creating Elements

JavaScript dapat membuat elemen HTML baru dan menambahkannya ke DOM secara dinamis.

createElement dan appendChild

// Membuat elemen baru
let newDiv = document.createElement('div');
let newP = document.createElement('p');
let newImg = document.createElement('img');

// Set content dan atribut
newP.textContent = 'Ini paragraf baru';
newP.className = 'new-paragraph';
newImg.src = 'image.jpg';
newImg.alt = 'Gambar baru';

// Menambahkan ke DOM
let container = document.querySelector('.container');
container.appendChild(newDiv);
container.appendChild(newP);
newDiv.appendChild(newImg);

insertBefore dan insertAdjacentElement

let container = document.querySelector('.container');
let existingElement = document.querySelector('.existing');
let newElement = document.createElement('div');
newElement.textContent = 'Element baru';

// insertBefore
container.insertBefore(newElement, existingElement);

// insertAdjacentElement
let anotherNew = document.createElement('span');
existingElement.insertAdjacentElement('beforebegin', anotherNew); // sebelum element
existingElement.insertAdjacentElement('afterbegin', anotherNew);  // di awal dalam element
existingElement.insertAdjacentElement('beforeend', anotherNew);   // di akhir dalam element
existingElement.insertAdjacentElement('afterend', anotherNew);    // setelah element

innerHTML vs createElement

let container = document.querySelector('.container');

// Cara 1: innerHTML (lebih mudah, tapi kurang aman)
container.innerHTML += `
    <div class="card">
        <h3>Judul Card</h3>
        <p>Deskripsi card</p>
        <button>Klik Saya</button>
    </div>
`;

// Cara 2: createElement (lebih aman, lebih kontrol)
function createCard(title, description) {
    let card = document.createElement('div');
    card.className = 'card';
    
    let h3 = document.createElement('h3');
    h3.textContent = title;
    
    let p = document.createElement('p');
    p.textContent = description;
    
    let button = document.createElement('button');
    button.textContent = 'Klik Saya';
    button.addEventListener('click', () => {
        alert('Button diklik!');
    });
    
    card.appendChild(h3);
    card.appendChild(p);
    card.appendChild(button);
    
    return card;
}

let newCard = createCard('Judul Baru', 'Deskripsi baru');
container.appendChild(newCard);

Menghapus Elemen

let elementToRemove = document.querySelector('.remove-me');

// Cara 1: remove() (modern)
elementToRemove.remove();

// Cara 2: removeChild() (lama)
let parent = elementToRemove.parentNode;
parent.removeChild(elementToRemove);

// Menghapus semua child elements
let container = document.querySelector('.container');
while (container.firstChild) {
    container.removeChild(container.firstChild);
}

// Atau dengan innerHTML
container.innerHTML = '';

Cloning Elements

let originalElement = document.querySelector('.template');

// Clone tanpa child elements
let shallowClone = originalElement.cloneNode(false);

// Clone dengan semua child elements
let deepClone = originalElement.cloneNode(true);

// Modifikasi clone
deepClone.id = 'cloned-element';
deepClone.querySelector('.title').textContent = 'Judul Clone';

// Tambahkan ke DOM
document.body.appendChild(deepClone);
