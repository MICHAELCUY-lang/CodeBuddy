Pelajaran 1: Pengenalan Events

Events adalah tindakan atau kejadian yang terjadi di browser, seperti klik mouse, penekanan keyboard, atau loading halaman.

Apa itu Events?

Events memungkinkan JavaScript untuk:

  • Merespons interaksi pengguna
  • Menjalankan kode saat kejadian tertentu
  • Membuat website yang interaktif
  • Validasi form secara real-time
  • Animasi dan efek visual

Jenis-jenis Events

// Mouse Events
// click, dblclick, mousedown, mouseup, mouseover, mouseout

// Keyboard Events  
// keydown, keyup, keypress

// Form Events
// submit, change, input, focus, blur

// Window Events
// load, resize, scroll, unload

// Touch Events (mobile)
// touchstart, touchend, touchmove

Event Handler Inline

<!-- HTML dengan inline event handler -->
<button onclick="alert('Button diklik!')">Klik Saya</button>

<button onclick="changeText()">Ubah Teks</button>

<script>
function changeText() {
    document.getElementById('demo').innerHTML = 'Teks berubah!';
}
</script>

Event Handler Property

// Menggunakan property event handler
let button = document.getElementById('myButton');

// Assign function ke event property
button.onclick = function() {
    alert('Button diklik!');
};

// Atau dengan arrow function
button.onclick = () => {
    console.log('Button diklik dengan arrow function');
};

// Assign function yang sudah ada
function handleClick() {
    alert('Function terpisah');
}

button.onclick = handleClick;

Event Object

// Event object berisi informasi tentang event
button.onclick = function(event) {
    console.log('Event type:', event.type); // 'click'
    console.log('Target element:', event.target); // element yang diklik
    console.log('Mouse X:', event.clientX); // posisi X mouse
    console.log('Mouse Y:', event.clientY); // posisi Y mouse
    console.log('Timestamp:', event.timeStamp); // waktu event
};

// Shorthand dengan arrow function
button.onclick = (e) => {
    console.log('Clicked element:', e.target.tagName);
};

Pelajaran 2: Event Listeners

addEventListener() adalah cara modern dan fleksibel untuk menangani events di JavaScript.

addEventListener Syntax

// Syntax: element.addEventListener(event, function, options)
let button = document.getElementById('myButton');

// Basic usage
button.addEventListener('click', function() {
    alert('Button diklik!');
});

// Dengan arrow function
button.addEventListener('click', () => {
    console.log('Arrow function event listener');
});

// Dengan function reference
function handleButtonClick() {
    alert('Function terpisah');
}

button.addEventListener('click', handleButtonClick);

Multiple Event Listeners

let button = document.getElementById('myButton');

// Bisa menambahkan multiple listeners untuk event yang sama
button.addEventListener('click', function() {
    console.log('Listener 1');
});

button.addEventListener('click', function() {
    console.log('Listener 2');
});

button.addEventListener('click', function() {
    console.log('Listener 3');
});

// Semua akan dijalankan saat button diklik

// Multiple events untuk element yang sama
button.addEventListener('mouseenter', function() {
    this.style.backgroundColor = 'lightblue';
});

button.addEventListener('mouseleave', function() {
    this.style.backgroundColor = '';
});

removeEventListener

// Untuk remove listener, harus menggunakan function reference
function myClickHandler() {
    console.log('Button diklik');
}

// Add listener
button.addEventListener('click', myClickHandler);

// Remove listener
button.removeEventListener('click', myClickHandler);

// Anonymous function tidak bisa di-remove
button.addEventListener('click', function() {
    console.log('Tidak bisa di-remove');
});

// Solusi: simpan reference
let clickHandler = function() {
    console.log('Bisa di-remove');
};

button.addEventListener('click', clickHandler);
button.removeEventListener('click', clickHandler);

Event Options

let button = document.getElementById('myButton');

// Options object
button.addEventListener('click', function() {
    console.log('Hanya sekali');
}, {
    once: true,      // Hanya dijalankan sekali
    passive: false,  // Bisa preventDefault
    capture: false   // Event bubbling phase
});

// Shorthand untuk once
button.addEventListener('click', function() {
    console.log('Sekali lagi');
}, { once: true });

// Passive listener (untuk performance)
document.addEventListener('scroll', function() {
    console.log('Scrolling...');
}, { passive: true });

Event Delegation

// Event delegation - attach listener ke parent
let container = document.getElementById('buttonContainer');

container.addEventListener('click', function(e) {
    // Cek apakah yang diklik adalah button
    if (e.target.tagName === 'BUTTON') {
        console.log('Button diklik:', e.target.textContent);
    }
});

// Berguna untuk dynamic content
function addNewButton() {
    let newButton = document.createElement('button');
    newButton.textContent = 'Button Baru';
    container.appendChild(newButton);
    // Tidak perlu add listener, sudah ter-handle oleh delegation
}

// Advanced delegation dengan class check
container.addEventListener('click', function(e) {
    if (e.target.classList.contains('delete-btn')) {
        e.target.parentElement.remove();
    }
    
    if (e.target.classList.contains('edit-btn')) {
        // Handle edit
    }
});

Pelajaran 3: Mouse Events

Mouse events memungkinkan kita merespons berbagai interaksi mouse seperti klik, hover, dan drag.

Click Events

let element = document.getElementById('clickable');

// Single click
element.addEventListener('click', function(e) {
    console.log('Single click');
    console.log('Button yang diklik:', e.button); // 0=left, 1=middle, 2=right
});

// Double click
element.addEventListener('dblclick', function() {
    console.log('Double click');
});

// Mouse down dan up
element.addEventListener('mousedown', function() {
    console.log('Mouse button ditekan');
});

element.addEventListener('mouseup', function() {
    console.log('Mouse button dilepas');
});

// Context menu (right click)
element.addEventListener('contextmenu', function(e) {
    e.preventDefault(); // Mencegah context menu default
    console.log('Right click detected');
});

Mouse Movement Events

let element = document.getElementById('hover-area');

// Mouse enter dan leave
element.addEventListener('mouseenter', function() {
    this.style.backgroundColor = 'lightblue';
    console.log('Mouse masuk area');

element.addEventListener('mouseleave', function() {
    this.style.backgroundColor = '';
    console.log('Mouse keluar area');

// Mouse over dan out (berbeda dengan enter/leave)
element.addEventListener('mouseover', function(e) {
    console.log('Mouse over:', e.target.tagName);
});

element.addEventListener('mouseout', function(e) {
    console.log('Mouse out:', e.target.tagName);

// Mouse move - tracking posisi mouse
element.addEventListener('mousemove', function(e) {
    let rect = this.getBoundingClientRect();
    let x = e.clientX - rect.left;
    let y = e.clientY - rect.top;
    
    console.log(`Mouse position: (${x}, ${y})`);
    
    // Update display
    document.getElementById('coordinates').textContent = `X: ${x}, Y: ${y}`;
});

Drag and Drop

// Simple drag implementation
let draggable = document.getElementById('draggable');
let isDragging = false;
let startX, startY, initialLeft, initialTop;

draggable.addEventListener('mousedown', function(e) {
    isDragging = true;
    startX = e.clientX;
    startY = e.clientY;
    
    let rect = this.getBoundingClientRect();
    initialLeft = rect.left;
    initialTop = rect.top;
    
    this.style.cursor = 'grabbing';
});

document.addEventListener('mousemove', function(e) {
    if (!isDragging) return;
    
    let deltaX = e.clientX - startX;
    let deltaY = e.clientY - startY;
    
    draggable.style.left = (initialLeft + deltaX) + 'px';
    draggable.style.top = (initialTop + deltaY) + 'px';
});

document.addEventListener('mouseup', function() {
    if (isDragging) {
        isDragging = false;
        draggable.style.cursor = 'grab';
    }
});

// HTML5 Drag and Drop API
let dragSource = document.getElementById('dragSource');
let dropZone = document.getElementById('dropZone');

dragSource.addEventListener('dragstart', function(e) {
    e.dataTransfer.setData('text/plain', this.id);
});

dropZone.addEventListener('dragover', function(e) {
    e.preventDefault(); // Allow drop
});

dropZone.addEventListener('drop', function(e) {
    e.preventDefault();
    let draggedId = e.dataTransfer.getData('text/plain');
    let draggedElement = document.getElementById(draggedId);
    this.appendChild(draggedElement);
});

    console.log('Ctrl pressed:', e.ctrlKey);
    console.log('Shift pressed:', e.shiftKey);

    // Mouse button

    console.log('Buttons:', e.buttons); // Bitmask of pressed buttons
});

Pelajaran 4: Keyboard Events

Keyboard events memungkinkan kita merespons input keyboard untuk membuat shortcut, validasi, dan interaksi yang lebih baik.

Jenis Keyboard Events

let input = document.getElementById('textInput');

// keydown - saat key ditekan (sebelum karakter muncul)
input.addEventListener('keydown', function(e) {
    console.log('Key down:', e.key);
    console.log('Key code:', e.keyCode); // Deprecated
    console.log('Which:', e.which); // Deprecated
    console.log('Code:', e.code); // Physical key
});

// keyup - saat key dilepas
input.addEventListener('keyup', function(e) {
    console.log('Key up:', e.key);
    console.log('Input value:', this.value);
});

// keypress - saat karakter dihasilkan (deprecated)
input.addEventListener('keypress', function(e) {
    console.log('Key press:', e.key);
    // Hanya untuk printable characters
});

Key Properties

document.addEventListener('keydown', function(e) {
    // e.key - nilai karakter yang dihasilkan
    console.log('Key:', e.key); // 'a', 'A', 'Enter', 'Escape', etc.
    
    // e.code - kode fisik key
    console.log('Code:', e.code); // 'KeyA', 'Enter', 'Escape', etc.
    
    // Modifier keys
    console.log('Ctrl:', e.ctrlKey);
    console.log('Shift:', e.shiftKey);
    console.log('Alt:', e.altKey);
    console.log('Meta:', e.metaKey);
    
    // Kombinasi keys
    if (e.ctrlKey && e.key === 's') {
        e.preventDefault();
        console.log('Ctrl+S pressed - Save action');
    }
    
    if (e.ctrlKey && e.key === 'z') {
        e.preventDefault();
        console.log('Ctrl+Z pressed - Undo action');
    }
});

Keyboard Shortcuts

// Global keyboard shortcuts
document.addEventListener('keydown', function(e) {
    // Escape key
    if (e.key === 'Escape') {
        closeModal();
    }
    
    // Arrow keys navigation
    switch(e.key) {
        case 'ArrowUp':
            navigateUp();
            break;
        case 'ArrowDown':
            navigateDown();
            break;
        case 'ArrowLeft':
            navigateLeft();
            break;
        case 'ArrowRight':
            navigateRight();
            break;
    }
    
    // Function keys
    if (e.key === 'F1') {
        e.preventDefault();
        showHelp();
    }
    
    // Number keys
    if (e.key >= '1' && e.key <= '9') {
        let tabIndex = parseInt(e.key) - 1;
        switchToTab(tabIndex);
    }
});

// Keyboard shortcuts untuk specific element
let editor = document.getElementById('editor');

editor.addEventListener('keydown', function(e) {
    // Ctrl+B untuk bold
    if (e.ctrlKey && e.key === 'b') {
        e.preventDefault();
        toggleBold();
    }
    
    // Ctrl+I untuk italic
    if (e.ctrlKey && e.key === 'i') {
        e.preventDefault();
        toggleItalic();
    }
    
    // Tab untuk indentation
    if (e.key === 'Tab') {
        e.preventDefault();
        insertTab();
    }
});

Input Validation

// Hanya allow angka
let numberInput = document.getElementById('numberOnly');

numberInput.addEventListener('keydown', function(e) {
    // Allow: backspace, delete, tab, escape, enter
    if ([8, 9, 27, 13, 46].indexOf(e.keyCode) !== -1 ||
        // Allow: Ctrl+A, Ctrl+C, Ctrl+V, Ctrl+X
        (e.keyCode === 65 && e.ctrlKey === true) ||
        (e.keyCode === 67 && e.ctrlKey === true) ||
        (e.keyCode === 86 && e.ctrlKey === true) ||
        (e.keyCode === 88 && e.ctrlKey === true) ||
        // Allow: home, end, left, right
        (e.keyCode >= 35 && e.keyCode <= 39)) {
        return;
    }
    
    // Ensure that it is a number and stop the keypress
    if ((e.shiftKey || (e.keyCode < 48 || e.keyCode > 57)) && 
        (e.keyCode < 96 || e.keyCode > 105)) {
        e.preventDefault();
    }
});

// Modern approach dengan input event
numberInput.addEventListener('input', function(e) {
    // Remove non-numeric characters
    this.value = this.value.replace(/[^0-9]/g, '');
});

// Real-time validation
let emailInput = document.getElementById('email');

emailInput.addEventListener('input', function(e) {
    let email = this.value;
    let isValid = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);

        this.style.borderColor = 'green';

        this.style.borderColor = 'red';
    }

Character Counter

let textarea = document.getElementById('message');
let counter = document.getElementById('charCounter');

// Prevent typing when limit reached
textarea.addEventListener('keydown', function(e) {
    if (this.value.length >= maxLength) {
        // Allow backspace, delete, arrow keys

            e.preventDefault();
        }
    }
});

Pelajaran 5: Form Events

Form events memungkinkan kita menangani interaksi dengan form elements seperti input, select, dan form submission.

Form Submission

let form = document.getElementById('myForm');

// Handle form submission
form.addEventListener('submit', function(e) {
    e.preventDefault(); // Mencegah default form submission
    
    // Validasi form
    let isValid = validateForm();
    
    if (isValid) {
        // Submit form via AJAX atau process data
        submitFormData();
    } else {
        showErrorMessage('Please fill all required fields');
    }
});

function validateForm() {
    let name = document.getElementById('name').value;
    let email = document.getElementById('email').value;
    let message = document.getElementById('message').value;
    
    if (!name || !email || !message) {
        return false;
    }
    
    // Email validation
    let emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
        return false;
    }
    
    return true;
}

function submitFormData() {
    let formData = new FormData(form);
    
    // Example AJAX submission
    fetch('/submit', {
        method: 'POST',
        body: formData
    })
    .then(response => response.json())
    .then(data => {
        showSuccessMessage('Form submitted successfully!');
        form.reset();
    })
    .catch(error => {
        showErrorMessage('Error submitting form');
    });
}

// Custom validation
function setupCustomValidation() {
    let passwordInput = document.getElementById('password');
    let confirmPasswordInput = document.getElementById('confirmPassword');
    
    confirmPasswordInput.addEventListener('input', function() {
        if (this.value !== passwordInput.value) {
            this.setCustomValidity('Passwords do not match');
        } else {
            this.setCustomValidity('');
        }
    });
    
    passwordInput.addEventListener('input', function() {
        let password = this.value;
        let strength = calculatePasswordStrength(password);
        
        updatePasswordStrengthIndicator(strength);
        
        if (strength < 3) {
            this.setCustomValidity('Password is too weak');
        } else {
            this.setCustomValidity('');
        }
    });
}

function calculatePasswordStrength(password) {
    let strength = 0;
    
    if (password.length >= 8) strength++;
    if (/[a-z]/.test(password)) strength++;
    if (/[A-Z]/.test(password)) strength++;
    if (/[0-9]/.test(password)) strength++;
    if (/[^A-Za-z0-9]/.test(password)) strength++;
    
    return strength;
}

// Real-time form validation
function setupRealTimeValidation() {
    let inputs = form.querySelectorAll('input, select, textarea');
    
    inputs.forEach(input => {
        input.addEventListener('blur', function() {
            validateField(this);
        });
        
        input.addEventListener('input', function() {
            if (this.classList.contains('error')) {
                validateField(this);
            }
        });
    });
}

function validateField(field) {
    let isValid = field.checkValidity();
    let errorElement = field.nextElementSibling;
    
    if (isValid) {
        field.classList.remove('error');
        if (errorElement && errorElement.classList.contains('error-message')) {
            errorElement.remove();
        }
    } else {
        field.classList.add('error');
        if (!errorElement || !errorElement.classList.contains('error-message')) {
            let errorMsg = document.createElement('span');
            errorMsg.className = 'error-message';
            errorMsg.textContent = field.validationMessage;
            field.parentNode.insertBefore(errorMsg, field.nextSibling);
        }
    }
}
