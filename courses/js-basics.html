Pelajaran 1: Pengenalan JavaScript

JavaScript adalah bahasa pemrograman yang digunakan untuk membuat halaman web menjadi interaktif dan dinamis.

Apa itu JavaScript?

JavaScript memungkinkan Anda untuk:

  • Memanipulasi konten HTML dan CSS
  • Merespons interaksi pengguna (klik, hover, dll)
  • Memvalidasi form
  • Membuat animasi dan efek visual
  • Berkomunikasi dengan server

Cara Menambahkan JavaScript

1. Inline JavaScript

<button onclick="alert('Hello World!')">Klik Saya</button>

2. Internal JavaScript

<script>
    console.log('Hello World!');
    alert('Selamat datang di JavaScript!');
</script>

3. External JavaScript

<script src="script.js"></script>

Console dan Debugging

console.log('Pesan debug');
console.error('Pesan error');
console.warn('Pesan warning');

Pelajaran 2: Variabel dan Tipe Data

Variabel adalah wadah untuk menyimpan data, dan JavaScript memiliki berbagai tipe data.

Deklarasi Variabel

// var (lama, hindari penggunaan)
var nama = 'John';

// let (untuk variabel yang bisa berubah)
let umur = 25;
umur = 26; // bisa diubah

// const (untuk konstanta)
const PI = 3.14159;
// PI = 3.14; // Error! tidak bisa diubah

Tipe Data Primitif

// String (teks)
let nama = 'Alice';
let pesan = "Hello World";
let template = `Halo, ${nama}!`; // Template literal

// Number (angka)
let umur = 25;
let tinggi = 175.5;
let negatif = -10;

// Boolean (true/false)
let aktif = true;
let selesai = false;

// Undefined
let belumDiisi;
console.log(belumDiisi); // undefined

// Null
let kosong = null;

Mengecek Tipe Data

console.log(typeof 'Hello'); // "string"
console.log(typeof 42); // "number"
console.log(typeof true); // "boolean"
console.log(typeof undefined); // "undefined"
console.log(typeof null); // "object" (ini bug di JavaScript!)

Pelajaran 3: Operator dan Ekspresi

Operator digunakan untuk melakukan operasi pada variabel dan nilai.

Operator Aritmatika

let a = 10;
let b = 3;

console.log(a + b); // 13 (penjumlahan)
console.log(a - b); // 7 (pengurangan)
console.log(a * b); // 30 (perkalian)
console.log(a / b); // 3.333... (pembagian)
console.log(a % b); // 1 (modulus/sisa bagi)
console.log(a ** b); // 1000 (pangkat)

Operator Perbandingan

let x = 5;
let y = '5';

console.log(x == y);  // true (sama nilai)
console.log(x === y); // false (sama nilai dan tipe)
console.log(x != y);  // false (tidak sama nilai)
console.log(x !== y); // true (tidak sama nilai atau tipe)
console.log(x > 3);   // true
console.log(x <= 5);  // true

Operator Logika

let benar = true;
let salah = false;

console.log(benar && salah); // false (AND)
console.log(benar || salah); // true (OR)
console.log(!benar);         // false (NOT)

// Short-circuit evaluation
let nama = '' || 'Default'; // 'Default'
let user = {name: 'John'} && user.name; // 'John'

Operator Assignment

let angka = 10;

angka += 5;  // angka = angka + 5 (15)
angka -= 3;  // angka = angka - 3 (12)
angka *= 2;  // angka = angka * 2 (24)
angka /= 4;  // angka = angka / 4 (6)
angka++;     // angka = angka + 1 (7)
angka--;     // angka = angka - 1 (6)

Pelajaran 4: Kondisi dan Perulangan

Struktur kontrol memungkinkan program membuat keputusan dan mengulang tindakan.

If Statement

let umur = 18;

if (umur >= 18) {
    console.log('Anda sudah dewasa');
} else if (umur >= 13) {
    console.log('Anda remaja');
} else {
    console.log('Anda masih anak-anak');
}

// Ternary operator
let status = umur >= 18 ? 'Dewasa' : 'Belum dewasa';

Switch Statement

let hari = 'Senin';

switch (hari) {
    case 'Senin':
        console.log('Awal minggu');
        break;
    case 'Jumat':
        console.log('Akhir minggu kerja');
        break;
    case 'Sabtu':
    case 'Minggu':
        console.log('Weekend!');
        break;
    default:
        console.log('Hari biasa');
}

For Loop

// For loop biasa
for (let i = 0; i < 5; i++) {
    console.log('Iterasi ke-' + i);
}

// For...of (untuk array)
let buah = ['apel', 'jeruk', 'pisang'];
for (let item of buah) {
    console.log(item);
}

// For...in (untuk object)
let person = {nama: 'John', umur: 30};

let i = 0;
while (i < 3) {

do {

} while (j < 3);

Pelajaran 5: Fungsi

Fungsi adalah blok kode yang dapat digunakan kembali untuk melakukan tugas tertentu.

Deklarasi Fungsi

// Function declaration
function sapa(nama) {
    return 'Halo, ' + nama + '!';
}

// Function expression
const tambah = function(a, b) {
    return a + b;
};

// Arrow function
const kali = (a, b) => a * b;
const kuadrat = x => x * x;
const salam = () => 'Hello World!';

Parameter dan Argument

// Default parameter
function perkenalan(nama = 'Anonim', umur = 0) {
    return `Nama: ${nama}, Umur: ${umur}`;
}

console.log(perkenalan()); // "Nama: Anonim, Umur: 0"
console.log(perkenalan('Alice', 25)); // "Nama: Alice, Umur: 25"

// Rest parameter
function jumlahSemua(...angka) {
    return angka.reduce((total, num) => total + num, 0);
}

console.log(jumlahSemua(1, 2, 3, 4)); // 10

Scope dan Closure

// Global scope
let globalVar = 'Saya global';

function luarFungsi() {
    // Function scope
    let lokalVar = 'Saya lokal';
    
    function dalamFungsi() {
        // Closure - bisa akses variabel parent
        console.log(globalVar); // OK
        console.log(lokalVar);  // OK
    }
    
    return dalamFungsi;
}

const fungsiClosure = luarFungsi();
fungsiClosure(); // Masih bisa akses lokalVar!

Pelajaran 6: Array dan Object

Array dan Object adalah struktur data penting untuk menyimpan koleksi data.

Array

// Membuat array
let buah = ['apel', 'jeruk', 'pisang'];
let angka = [1, 2, 3, 4, 5];
let campuran = ['teks', 42, true, null];

// Mengakses elemen
console.log(buah[0]); // 'apel'
console.log(buah.length); // 3

// Menambah elemen
buah.push('mangga'); // tambah di akhir
buah.unshift('anggur'); // tambah di awal

// Menghapus elemen
buah.pop(); // hapus dari akhir
buah.shift(); // hapus dari awal

Array Methods

let angka = [1, 2, 3, 4, 5];

// Map - transform setiap elemen

// Map - transform setiap elemen
let dikali2 = angka.map(x => x * 2); // [2, 4, 6, 8, 10]

// Filter - saring elemen
let genap = angka.filter(x => x % 2 === 0); // [2, 4]

// Find - cari elemen pertama
let ditemukan = angka.find(x => x > 3); // 4

// Reduce - gabungkan menjadi satu nilai
let total = angka.reduce((sum, x) => sum + x, 0); // 15

// ForEach - iterasi
angka.forEach(x => console.log(x));

Object

// Membuat object
let person = {
    nama: 'John',
    umur: 30,
    hobi: ['membaca', 'coding'],
    alamat: {
        kota: 'Jakarta',
        kodePos: '12345'
    }
};

// Mengakses properti
console.log(person.nama); // 'John'
console.log(person['umur']); // 30
console.log(person.alamat.kota); // 'Jakarta'

// Menambah/mengubah properti
person.pekerjaan = 'Developer';
person.umur = 31;

Object Methods

let mobil = {
    merk: 'Toyota',
    model: 'Camry',
    tahun: 2020,
    
    // Method
    info: function() {
        return `${this.merk} ${this.model} (${this.tahun})`;
    },
    
    // Arrow function tidak memiliki 'this'
    infoArrow: () => {
        return 'Arrow function';
    }
};

console.log(mobil.info()); // "Toyota Camry (2020)"

// Object methods
console.log(Object.keys(mobil)); // ['merk', 'model', 'tahun', 'info']
console.log(Object.values(mobil)); // ['Toyota', 'Camry', 2020, function...]
